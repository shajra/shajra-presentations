```tut:silent
type StateEff[F[_]] = MonadState[Int, F]
type ErrorEff[F[_]] = MonadError[String, F]

def tryIt[F[_] : ErrorEff : StateEff] = {
  val attempt =
    for {
      _ <- 1.put[F]
      _ <- "oh noes".raiseError[F, Unit]
    } yield ()
  attempt.handleError(_ => ().pure[F])
}
```
